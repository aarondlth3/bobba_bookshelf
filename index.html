/** * ============================================================================== * CONFIGURACIÓN INICIAL * ============================================================================== * Se ha configurado automáticamente el ID de la carpeta de Google Drive * que proporcionaste ("bobba_bookshelf"). */ const FOLDER_ID = "1bUA8eEsYcxVVrCFCE1MK1gOQAd4FvDOl"; /** * 1. MÉTODO GET (doGet) * Retorna la lista de audios o resuelve la redirección de streaming en tiempo real. */ function doGet(e) { try { const action = e.parameter.action; // --- NUEVA ACCIÓN: RESOLVER REDIRECCIÓN DE STREAMING EN TIEMPO REAL --- if (action === "get_stream_url") { const driveId = e.parameter.driveId; if (!driveId) { return construirRespuestaJson({ success: false, error: "Falta el ID de Drive." }); } const directUrl = resolverUrlDirectaCdn(driveId); return construirRespuestaJson({ success: true, url: directUrl }); } // --- ACCIÓN POR DEFECTO: LISTAR AUDIOS --- const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = sheet.getDataRange().getValues(); const responseData = []; if (data.length > 1) { for (let i = 1; i < data.length; i++) { const id = data[i][0]; const nombre = data[i][1]; const driveId = data[i][2]; const streamUrl = `https://drive.google.com/uc?export=download&id=${driveId}`; responseData.push({ id: id, nombre: nombre, driveId: driveId, url: streamUrl }); } } return construirRespuestaJson({ success: true, audios: responseData }); } catch (error) { return construirRespuestaJson({ success: false, error: "Error al procesar solicitud GET: " + error.toString() }); } } /** * FUNCIÓN AUXILIAR: resolverUrlDirectaCdn * Hace una petición simulada en el servidor para seguir las redirecciones de Google Drive * y obtener el enlace directo final sin redirecciones, garantizando compatibilidad con iOS/Safari. */ function resolverUrlDirectaCdn(driveId) { const urlUrl = `https://drive.google.com/uc?export=download&id=${driveId}`; try { // Primera petición: detecta la redirección de drive.google.com const response = UrlFetchApp.fetch(urlUrl, { followRedirects: false, muteHttpExceptions: true, headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36' } }); const headers = response.getHeaders(); let location = headers['Location'] || headers['location']; if (location) { // Segunda petición: si redirige a drive.usercontent, resolvemos para obtener el nodo CDN directo if (location.includes('drive.usercontent.google.com') || location.includes('google.com')) { const secondResponse = UrlFetchApp.fetch(location, { followRedirects: false, muteHttpExceptions: true, headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36' } }); const secondHeaders = secondResponse.getHeaders(); const secondLocation = secondHeaders['Location'] || secondHeaders['location']; if (secondLocation) { location = secondLocation; } } return location; } } catch (e) { // Si algo falla, retornamos la URL de fallback convencional } return urlUrl; } /** * 2. MÉTODO POST (doPost) * Soporta carga fragmentada de alta velocidad para audiolibros grandes. */ function doPost(e) { try { const postData = JSON.parse(e.postData.contents); const action = postData.action; const folder = DriveApp.getFolderById(FOLDER_ID); // --- ACCIÓN A: GUARDAR FRAGMENTO TEMPORAL --- if (action === "upload_chunk") { const uploadId = postData.uploadId; const chunkIndex = postData.chunkIndex; const base64Data = postData.fileData; const tempFileName = `_temp_${uploadId}_${chunkIndex}.tmp`; folder.createFile(tempFileName, base64Data, MimeType.PLAIN_TEXT); return construirRespuestaJson({ success: true, message: `Fragmento ${chunkIndex} guardado.` }); } // --- ACCIÓN B: UNIR TODOS LOS FRAGMENTOS --- if (action === "merge_chunks") { const uploadId = postData.uploadId; const fileName = postData.fileName; const totalChunks = parseInt(postData.totalChunks); let combinedBase64 = ""; for (let i = 0; i < totalChunks; i++) { const tempFileName = `_temp_${uploadId}_${i}.tmp`; const files = folder.getFilesByName(tempFileName); if (files.hasNext()) { const tempFile = files.next(); combinedBase64 += tempFile.getBlob().getDataAsString(); tempFile.setTrashed(true); // Limpiar espacio temporal } else { throw new Error(`Falta el fragmento temporal número ${i}. Recomienza la subida.`); } } const contentType = "audio/mp4"; // Estándar seguro para m4a const decodedData = Utilities.base64Decode(combinedBase64); const blob = Utilities.newBlob(decodedData, contentType, fileName); const finalFile = folder.createFile(blob); // Aplicar permisos globales de lectura para el streaming correcto finalFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); const fileId = finalFile.getId(); const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const idUnico = "AUD-" + new Date().getTime(); sheet.appendRow([idUnico, fileName, fileId]); return construirRespuestaJson({ success: true, id: idUnico, nombre: fileName, driveId: fileId, message: "Audio reconstruido correctamente." }); } // --- ACCIÓN C: COMPATIBILIDAD CON SUBIDAS SIMPLES --- const base64Data = postData.fileData; const fileName = postData.fileName; if (!base64Data || !fileName) { return construirRespuestaJson({ success: false, error: "Acción no reconocida en el servidor." }); } const decodedData = Utilities.base64Decode(base64Data); const blob = Utilities.newBlob(decodedData, "audio/mp4", fileName); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); const fileId = file.getId(); const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const idUnico = "AUD-" + new Date().getTime(); sheet.appendRow([idUnico, fileName, fileId]); return construirRespuestaJson({ success: true, id: idUnico, nombre: fileName, driveId: fileId }); } catch (error) { return construirRespuestaJson({ success: false, error: "Error interno del servidor: " + error.toString() }); } } /** * Emite respuestas con compatibilidad CORS para GitHub Pages. */ function construirRespuestaJson(objeto) { const jsonString = JSON.stringify(objeto); return ContentService.createTextOutput(jsonString) .setMimeType(ContentService.MimeType.JSON); }
